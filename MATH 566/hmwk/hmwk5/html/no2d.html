function [A,x,f] = algorithm(n)
    %coefficients
    i = [1:n]'; a = i;
    j = [1:n-1]'; b = -(j+1)/3; c = b;
    k = [1:n-2]'; d = -(k+2)/6; e = d;

    %RHS
    l = [3:n-2]'; fi(l-2) = 0;
    f = [1/2 1/6 fi 1/6 1/2]';

    [x] = pentadiagonal(a,b,c,d,e,f);

    %Coefficient matrix
    A = diag(d,-2) + diag(b,-1) + diag(a) + diag(c,1) + diag(e,2);
end
